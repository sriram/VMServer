  def vm_mkdir(dir)
    vm_command = "#{@base_command} -gu #{@guest_user} -gp #{@guest_password} #{command} \'#{@datastore}\' \'#{dir}\'"
    log vm_command
    result = system(vm_command)
    result  ? log("Directory created successfully in guest.") : log("Error! Directory could not be created.")
    result
  end
