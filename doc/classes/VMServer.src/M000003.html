  def file_exists_in_guest?(file)
    vm_command = "#{@base_command} -gu #{@guest_user} -gp #{@guest_password} fileExistsInGuest \'#{datastore}\' #{file}"
    output = system(vm_command)
    if output =~ /The file exists/
      return true
    else
      return false
    end
  end
